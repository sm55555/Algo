import java.io.*;
import java.util.*;

public class Main {
		static int n,m;
		static int[] a;
	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);
		
		n = sc.nextInt();
		m = sc.nextInt();
		
		a = new int[n+1];
		
		for (int i = 1; i <= n; i++) {
			a[i] = sc.nextInt();
			a[i] = a[i-1] + a[i];			
		}
		
		for (int i = 0; i < m; i++) {
			int a1 = sc.nextInt();
			int a2 = sc.nextInt();
			
			int res = a[a2]-a[a1-1];
			System.out.println(res);
		}
		
	}
}
