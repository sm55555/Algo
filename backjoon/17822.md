import java.util.ArrayList;
import java.util.LinkedList;
import java.util.Queue;
import java.util.Scanner;

public class Main17822 {
	static int[][] map = new int[50][50];
	static int[] dx = { -1, 0, 1, 0 };
	static int[] dy = { 0, -1, 0, 1 };
	static Queue<Point> q = new LinkedList<>();
	static int n, m, t;

	static class Point {
		int x;
		int y;
		int v;

		Point(int y, int x, int v) {
			this.x = x;
			this.y = y;
			this.v = v;
		}
	}

	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);
		n = sc.nextInt();
		m = sc.nextInt();
		t = sc.nextInt();

		for (int i = 0; i < n; i++) {
			for (int j = 0; j < m; j++) {
				int value = sc.nextInt();
				map[i][j] = value;
			}
		}
		
		for (int z = 0; z < t; z++) {
			// 먼저 원판 돌리기
			int line = sc.nextInt();
		    int dir = sc.nextInt();
		    int amount = sc.nextInt() % m;
		    
		    
		 // 원판 돌리기
		    LinkedList<Integer> q2 = new LinkedList<Integer>();

			for (int i = 0; i < n; i++) {
				q2 = new LinkedList<Integer>();
				if ((i + 1) % line == 0) {
					for (int j = 0; j < m; j++) {
						q2.add(map[i][j]);
					}

					// clock >>
					if (dir == 0) {
						for (int j = 0; j < amount; j++) {
							int zz = q2.removeLast();
							q2.addFirst(zz);
						}
					} else {
						// countclock <<
						for (int j = 0; j < amount; j++) {
							int zz = q2.removeFirst();
							q2.addLast(zz);
						}
					}

					for (int j = 0; j < m; j++) {
						map[i][j] = q2.get(j);
					}

				}
			}
			/// 원판 끝 
			 double psum = 0;
		      int cnt = 0;
		      for (int ii = 0; ii < n; ii++) {
		        for (int jj = 0; jj < m; jj++) {
		          if (map[ii][jj] > 0) {
		            q.add(new Point(ii, jj, map[ii][jj]));
		            psum += map[ii][jj];
		            cnt++;
		          }
		        }
		      }
			
		      if (!bfs()) {
		          // 인접한 것들이 같은 수가 아닐 때
		          psum = psum * 1.0 / cnt;
		          update(psum);
		        }
			
		}
		
		int sum = 0;
	    for (int i = 0; i < n; i++) {
	      for (int j = 0; j < m; j++) {
	        sum += map[i][j];
	      }
	    }
	    System.out.println(sum);

	

	}

	public static boolean bfs() {
	    int size = q.size();
	    boolean flag = false;
	    for (int i = 0; i < size; i++) {
	      Point p = q.poll();
	      for (int j = 0; j < 4; j++) {
	        int nx = dx[j] + p.x;
	        int ny = dy[j] + p.y;

	        if (j == 0 && p.x == 0) {
	          nx = m - 1;
	        }

	        if (j == 2 && p.x == m - 1) {
	          nx = 0;
	        }

	        if (0 <= nx && nx < m && 0 <= ny && ny < n) {
	          if (map[ny][nx] == p.v) {
	            flag = true;
	            map[p.y][p.x] = 0;
	            map[ny][nx] = 0;
	          }
	        }
	      }
	    }
	    return flag;
	  }

	public static void update(double sum) {
	    for (int i = 0; i < n; i++) {
	      for (int j = 0; j < m; j++) {
	        if (map[i][j] > 0) {
	          if (map[i][j] > sum) {
	            map[i][j] -= 1;
	          } else if (map[i][j] < sum) {
	            map[i][j] += 1;
	          }
	        }
	      }
	    }
	  }
}
