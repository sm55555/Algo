import java.util.Arrays;
import java.util.Scanner;

public class Main {
		static char[][] map;
		static int r,c,x,y;
	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);		
		r = sc.nextInt();
		c = sc.nextInt();
		
		map = new char[r][c];

	 	char[][] map2 = new char[2*r][2*c];
		
		for (int i = 0; i < r; i++) {
			String s = sc.next();
			for (int j = 0; j < c; j++) {
				map[i][j] = s.charAt(j);
				map2[i][j] = map[i][j];
			}
		}
		
		x = sc.nextInt();
		y = sc.nextInt();
		
		// 가로 복제
		
	 	for (int i = 0; i < r; i++) {
	 		int temp = c-1;
			for (int j = c; j < 2*c; j++) {
				map2[i][j] = map[i][temp--];
			}
		}
	 	
	 	// 세로 복제
	 	int temp = r-1;
	 		for (int i = r; i < r*2; i++) {
	 			for (int j = 0; j < c*2; j++) {
	 				map2[i][j] = map2[temp][j];
	 			}
	 			temp--;
	 		}
	 	
	 	if(map2[x-1][y-1] =='.') {
	 		map2[x-1][y-1]='#';
	 	}else {
	 		map2[x-1][y-1]='.';
	 	}
	 		
	 	// 결과
		
		for (int i = 0; i < r*2; i++) {
			for (int j = 0; j < c*2; j++) {
				System.out.print(map2[i][j]);
			}
			System.out.println();
		}
		
		
	}

}
