public class Main {
	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);

		String s = sc.next();
		int ans =0;
		for (int i = 0; i < s.length(); i++) {
			int temp = s.charAt(i);
			
			if(temp >= 97) {
				temp -= 96;
			}else {
				temp -=38;
			}
			ans += temp;			
		}
		
		if(ans ==1) {
			System.out.println("It is a prime word.");
			return;
		}
		
		check(ans);
		
	}

	private static void check(int ans) {
		// TODO Auto-generated method stub

		}
