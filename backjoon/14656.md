```java

import java.util.Scanner;

public class Main {
		
	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);

		int n = sc.nextInt();
		int ans =0;
		for (int i = 1; i <=n; i++) {
			
			int temp = sc.nextInt();
			
			if(temp != i) {
				ans++;
			}
		}
		
		System.out.println(ans);
		
	}
}
```
