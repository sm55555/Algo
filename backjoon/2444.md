		// 5줄에 걸친다..

				System.out.print(" ");
			}
			
			for (int j = 1; j <=2*(i-1)+1; j++) {
				System.out.print("*");
			}
			
			System.out.println();
		}

		for (int i = 1; i <= n; i++) {
			
			for (int j = 0; j <i-1; j++) {

			}
			
			for (int j = 1; j <2*(n-i+1); j++) {
				System.out.print("*");
			}

	}


}

	for (int i = 1; i < n; i++) {
		for (int j = 1; j <= n-i; j++) {
			cout << " ";
		}

		// 별은 n=5일때  1  3 5 7 8 ..
		for (int j = 1; j <= 2*(i-1)+1; j++) {
			cout << "*";
		}

		cout << endl;
	}


	for (int i = 1; i <= n; i++) {
		
		for (int j = 1; j <i; j++) {
			cout << " ";
		}

		for (int j = 1; j<2*(n-i+1); j++) {
			cout << "*";
		}

		cout << endl;
	}
		
	return 0;

}
