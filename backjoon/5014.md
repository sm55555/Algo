import java.util.Arrays;
import java.util.LinkedList;
import java.util.Queue;
import java.util.Scanner;

public class Main {
	static int f,s,g,u,d;
	static int[] dx = new int[2];
	static boolean[] c;
	static class point{ 
		int cnt;
		int floor;
		
		point(int cnt, int floor){
			this.cnt = cnt;
			this.floor = floor;
		}
	}
	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);		
		
		//10 1 10 2 1
		f = sc.nextInt();
		s = sc.nextInt();
		g = sc.nextInt();
		u = sc.nextInt();
		d = sc.nextInt();
		c = new boolean[f+1];
		
		dx[0] = u;
		dx[1] = -d;
	
		Queue<point> q = new LinkedList();
		q.add(new point(0,s));
		
		while(!q.isEmpty()) {
			c[s]=true;
			
			point p = q.remove();
			
			int x = p.cnt;
			int y = p.floor;
			
			if(y == g) {
				System.out.println(x);
				return;
			}
			
			for (int i = 0; i < 2; i++) {
				int ny = y + dx[i];
				
				if(ny <1 || ny >f) {
					continue;
				}
								
				if(c[ny]) {
					continue;
				}			
				q.add(new point(x+1,ny));
				c[ny] = true;
			}
			
		}
		
		System.out.println("use the stairs");
		
		
		
	}


}
