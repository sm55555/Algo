import java.util.ArrayList;
import java.util.Arrays;
import java.util.Scanner;

public class Main {
		public static void main(String[] args){
			Scanner sc = new Scanner(System.in);
			String s = sc.next();
			
			for (int i = 0; i < s.length(); i++) {
				
				int a = s.charAt(i);
				
				// 65 -> 88
				// 66 -> 89
				
				if(a<=67) {
					a +=23;
				}else {
					a -=3;
				}
				
				
				
				char temp = (char)a;
				
				System.out.print(temp);
				
			}
			
			
		}
}
