import java.io.*;
import java.util.*;

public class Main {
	
	// 전위
	
	// 중위
	
	// 후위
	static ArrayList<Character> preorder;
	static ArrayList<Character> inorder;
	static ArrayList<Character> postorder;

	static class Node{
		char name;
		Node left;
		Node right;
	}
	
	static int n;
	
	public static void main(String[] args) throws IOException {

		Scanner sc = new Scanner(System.in);
		n = sc.nextInt();
		preorder = new ArrayList<Character>();
		inorder = new ArrayList<Character>();
		postorder = new ArrayList<Character>();
		ArrayList<Node> nodes = new ArrayList();
		
		for (int i = 0; i < n; i++) {
			Node node = new Node();
			node.name = (char)(i+65);
			nodes.add(node);
		}
		
		for (int i = 0; i < n; i++) {
			char node = (sc.next()).charAt(0);
			char left = (sc.next()).charAt(0);
			char right = (sc.next()).charAt(0);
			
			Node thisnode = nodes.get(node-65);
			if(left !='.') {
				Node leftnode = nodes.get(left-65);
				thisnode.left = leftnode;
			}
			
			if(right !='.') {
				Node rightnode = nodes.get(right-65);
				thisnode.right = rightnode;
			}
		}
		
		preorder(nodes.get(0));
		
		for(char c : preorder) {
			System.out.print(c);
		}
		System.out.println();
		
		inorder(nodes.get(0));
		
		for(char c : inorder) {
			System.out.print(c);
		}
		System.out.println();
		
		postorder(nodes.get(0));
		
		for(char c : postorder) {
			System.out.print(c); 
		}
		System.out.println();

	}

	private static void postorder(Node node) {
		// TODO Auto-generated method stub
		// 후위 순회는  왼쪽, 오른쪽, 루트
		
		if(node.left !=null){
			postorder(node.left);
		}
		
		if(node.right != null) {
			postorder(node.right);
		}
		
		postorder.add(node.name);
	}

	private static void inorder(Node node) {
		// TODO Auto-generated method stub
		// 중위 순회는 왼쪽, 루트, 오른쪽
		if(node.left != null) {
			inorder(node.left);
		}
		inorder.add(node.name);
		if(node.right != null) {
			inorder(node.right);
		}
		
	}

	private static void preorder(Node node) {
		// TODO Auto-generated method stub
		// 전위 순회는 루트, 왼, 오
		
		preorder.add(node.name);
		if(node.left != null) {
			preorder(node.left);			
		}
		if(node.right != null) {
			preorder(node.right);
		}
		
	}

}
