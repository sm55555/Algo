import java.util.Arrays;
import java.util.Scanner;

public class Main {
		static int n;
		static long[] a;
		static long maxValue;
		static int Cnt;
		static int maxCnt = Integer.MIN_VALUE;
	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);
		n = sc.nextInt();
		
		a = new long[n];
		
		for (int i = 0; i < n; i++) {
			a[i] = sc.nextLong();
		}
		
		Arrays.sort(a);
		
		maxValue = a[0];
		Cnt = 1;
		
		for (int i = 0; i < n-1; i++) {
			
			if(a[i] ==a[i+1]) {
				Cnt++;
				
				if(Cnt > maxCnt) {
					maxCnt = Cnt;
					maxValue = a[i];
				}
			}else {
				Cnt=1;				
			}

		}
		
		System.out.println(maxValue);
		
	}

}
